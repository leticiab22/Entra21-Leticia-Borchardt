Exercício 18)

        let R = Number(prompt("Forneça o valor do raio de uma esfera:"))
        console.log(`Raio: ${R}`)
        const pi = 3.14159
        let volume = (4/3) * pi * R ** 3
        console.log(`Volume: ${volume.toFixed(2)}`)
        alert(`Volume da esfera: ${volume.toFixed(2)}`)
